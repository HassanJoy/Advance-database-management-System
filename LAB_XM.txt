FROM EMP,DEPT,SALGRADE
WHERE EMP.DEPTNO=DEPT.DEPTNO AND SAL BETWEEN LOSAL AND HISAL
AND SALGRADE.GRADE=2 AND EMP.DEPTNO IN (10,20) AND JOB='ANALYST'

WHERE SAL BETWEEN LOSAL AND HISAL AND EMP.DEPTNO=DEPT.DEPTNO  AND EMP.DEPTNO=(SELECT DEPTNO FROM EMP
GROUP BY DEPTNO  HAVING COUNT(EMPNO)=(SELECT MAX(COUNT(EMPNO)) FROM EMP
GROUP BY DEPTNO));
